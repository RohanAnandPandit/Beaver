pi = 3.1415926535
e = 2.718281828
phi = 1.618

class Complex (
	def init {a, b} (
		this.real = a; this.imag = b
	)
		
	def toString = "(" ++ show real ++ "+" ++ show imag ++ "i)"

	def add (Complex num) = Complex {real + num.real, imag + num.imag}
	def add num = Complex {real + num, imag}

	def mul (Complex num) = (
		Complex {real * num.real - imag * num.imag, real * num.imag + imag * num.real}
	)
)

i = Complex {0, 1}
