class LinkedListNode (
	def LinkedListNode {data, next} (
		this.data = data
		this.next = next
	)

	def toString = show data ++ " -> (" ++ show next ++ ")"
)

class LinkedList (
	def LinkedList {} (
		this.front = ?
		this.back = ?
	)

	def add data (
		node = LinkedListNode {data, ?}
		if (front == ?) (
			this.front = node
			return ?
		)
		if (back == ?) (
			this.back = node
			this.front.next = node
			return ?
		)
		this.back.next = node
		this.back = node
	)

	def toString = show front
)
